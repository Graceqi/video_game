for d in data:
    print(d)
    x=video_data[d[0]]
    y=video_data[d[1]]
    #皮尔森相似度
    print("皮尔森相似度",pearsonr(x,y)[0])
    #余弦相似度计算方法
    tmp=sum(a*b for a,b in zip(x,y))
    non=np.linalg.norm(x)*np.linalg.norm(y)
    print("余弦相似度",round(tmp/float(non),3))
    #欧几里得相似度计算方法
    print("欧几里得相似度计算",math.sqrt(sum(pow(a-b,2) for a,b in zip(x,y))))
